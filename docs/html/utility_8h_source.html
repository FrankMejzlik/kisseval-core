114

116

123  }

135  continue;
136  }
137

140  ++depth;
141  continue;
142  }
143 
144  if (c == '+')

146  --depth;
147  if (depth == 0)
148  {
149  // Dispatch clause
150  result.emplace_back(clause_buffer);
151  clause_buffer = Clause();

153  continue;
154  }
